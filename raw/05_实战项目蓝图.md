### 2.1 离线索引 Pipeline

```go
// 完整索引流程
func BuildIndex(ctx context.Context, docPath string) error {
    // Step 1: 加载
    loader, _ := fileloader.NewFileLoader(ctx, &fileloader.Config{FilePath: docPath})
    docs, _ := loader.Load(ctx, document.Source{})

    // Step 2: 切分
    splitter, _ := textsplitter.NewTextSplitter(ctx, &textsplitter.Config{
        ChunkSize: 512, ChunkOverlap: 50,
    })
    chunks, _ := splitter.Transform(ctx, docs)

    // Step 3: Embedding + 存储
    embedder, _ := ark.NewEmbedder(ctx, &ark.EmbeddingConfig{
        APIKey: os.Getenv("ARK_API_KEY"),
        Model:  os.Getenv("EMBED_MODEL"),
    })
    indexer, _ := milvus2.NewIndexer(ctx, &milvus2.IndexerConfig{
        Address: "localhost:19530",
        CollectionName: "knowledge_base",
        Embedder: embedder,
    })
    return indexer.Store(ctx, chunks)
}
```

### 2.2 在线 Corrective RAG Graph

```
用户问题
  → 混合检索（向量 + BM25）
  → 相关性评分
    ├── 分数高 → 生成答案
    └── 分数低 → 重写查询 → 重新检索 → 生成答案
  → 幻觉检测
    ├── 无幻觉 → 输出
    └── 有幻觉 → 重新生成（最多2次）
```

```go
func BuildRAGAgent(ctx context.Context) (*compose.CompiledGraph, error) {
    g := compose.NewGraph[string, string]()

    g.AddLambdaNode("retrieve", hybridRetrieveLambda)
    g.AddLambdaNode("grade",    gradeLambda)
    g.AddLambdaNode("rewrite",  rewriteLambda)
    g.AddChatModelNode("generate", chatModel)
    g.AddLambdaNode("check",    hallucinationCheckLambda)

    g.AddEdge(compose.START, "retrieve")
    g.AddEdge("retrieve", "grade")
    g.AddBranch("grade", gradeBranch)         // relevant/irrelevant
    g.AddEdge("rewrite", "retrieve")          // 循环：重写后重检索
    g.AddEdge("generate", "check")
    g.AddBranch("check", hallucinationBranch) // ok/retry
    g.AddEdge("check", compose.END)

    return g.Compile(ctx, compose.WithMaxRunSteps(10))
}
```

### 2.3 暴露为 A2A Server

```go
type RAGAgentServer struct {
    agent *compose.CompiledGraph
}

func (s *RAGAgentServer) HandleSendMessage(w http.ResponseWriter, r *http.Request) {
    var req A2ARequest
    json.NewDecoder(r.Body).Decode(&req)

    // 提取用户消息
    query := extractText(req.Params.Message)

    // 调用 Eino RAG Agent
    result, err := s.agent.Invoke(r.Context(), query)

    // 包装为 A2A 响应
    task := A2ATask{
        ID:      req.Params.Message.TaskID,
        Status:  A2AStatus{State: "completed"},
        Artifacts: []A2AArtifact{
            {Parts: []A2APart{{Kind: "text", Text: result}}},
        },
    }
    json.NewEncoder(w).Encode(task)
}

func main() {
    server := &RAGAgentServer{agent: BuildRAGAgent(ctx)}

    // Agent Card
    http.HandleFunc("/.well-known/agent.json", serveAgentCard)
    // A2A 接口
    http.HandleFunc("/message/send",   server.HandleSendMessage)
    http.HandleFunc("/message/stream", server.HandleStreamMessage)

    http.ListenAndServe(":8080", nil)
}
```

## 3. Gateway Agent 设计

```go
// Gateway：接收用户请求，路由到合适的子 Agent（通过 A2A）
func routeRequest(ctx context.Context, query string) string {
    // 1. LLM 意图分类
    intent := classifyIntent(ctx, query)

    // 2. 根据意图选择 Agent
    var targetURL string
    switch intent {
    case "knowledge":  targetURL = "http://rag-agent:8080"
    case "search":     targetURL = "http://search-agent:8081"
    case "analysis":   targetURL = "http://analysis-agent:8082"
    }

    // 3. 通过 A2A 协议发送任务
    result := sendA2AMessage(ctx, targetURL, query)
    return result
}
```
